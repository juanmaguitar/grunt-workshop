Code

module.exports = function(grunt) {
    grunt.initConfig({
        concat: {
            main: {
                src: 'src/*.js',
                dest: 'build/scripts.js'
            }
        },
        min: {
            main: {
                src: '<config:concat.main.dest>',
                dest: 'build/scripts.min.js'
            }
        }
    });
    grunt.registerTask('default', 'concat min');
};
